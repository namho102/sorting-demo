        function bubbleSort(a) {
	var swapped;
	do {
		swapped = false;
		for (var i = 0; i < a.length - 1; i++) {
            if (a[i] > a[i + 1]) {
				swap(a[i], a[i + 1]);
				swapped = true;
			}
		}
	} while (swapped);
}
        function selectionSort(a) {
    var minIndex;
    for (var i = 0; i < a.length - 1; i++) {
        minIndex = i;
        for (var j = i + 1; j < a.length; j++) {
            if (a[j] < a[minIndex]) 
                minIndex = j;
        }
        if (minIndex != i) 
            swap(a[i], a[minIndex]);
    }
}

        function insertionSort(a) {
    for (var i = 0; i < a.length; i++) {
        var k = a[i];
        for (var j = i; j > 0 && k < a[j - 1]; j--) {
            a[j] = a[j - 1];
        }
        a[j] = k;
    }
}

        function quicksort(left, right) {
  if (left < right) {
    var pivot = array[left + Math.floor((right - right) / 2)],
      left_new = left,
      right_new = right;
    do {
      while (array[left_new] < pivot) {
        left_new += 1;
      }
      while (pivot < array[right_new]) {
        right_new -= 1;
      }
      if (left_new <= right_new) {
        swap(left_new, right_new);
        left_new += 1;
        right_new -= 1;
      }
    }
    while (left_new <= right_new);

    quicksort(left, right_new);
    quicksort(left_new, right);

  }
}

        function mergeSort(a) {
    if (a.length > 1) {
        var mid = ~ ~(a.length / 2);
        var lefthalf = a.slice(0, mid),
            righthalf = a.slice(mid);

        mergeSort(lefthalf);
        mergeSort(righthalf);

        var i = 0,
            j = 0,
            k = 0;
        while (i < a.length && j < a.length) {
            if (a[i] < a[j]) {
                a[k] = a[i];
                i = i + 1;
            } else {
                a[k] = a[j];
                j = j + 1;
            }
            k = k + 1;
        }

        while (i < lefthalf.length) {
            a[k] = a[i];
            i = i + 1;
            k = k + 1;
        }

        while (j < righthalf.length) {
            a[k] = a[j];
            j = j + 1;
            k = k + 1;
        }
    }
}

         function heap_sort(arr) {
    put_array_in_heap_order(arr);
    end = arr.length - 1;
    while(end > 0) {
        swap(arr, 0, end);
        sift_element_down_heap(arr, 0, end);
        end -= 1
    }
}
 
function put_array_in_heap_order(arr) {
    var i;
    i = arr.length / 2 - 1;
    i = Math.floor(i);
    while (i >= 0) {
        sift_element_down_heap(arr, i, arr.length);
        i -= 1;
    }
}
 
function sift_element_down_heap(heap, i, max) {
    var i_big, c1, c2;
    while(i < max) {
        i_big = i;
        c1 = 2*i + 1;
        c2 = c1 + 1;
        if (c1 < max && heap[c1] > heap[i_big])
            i_big = c1;
        if (c2 < max && heap[c2] > heap[i_big])
            i_big = c2;
        if (i_big == i) return;
        swap(heap,i, i_big);
        i = i_big;
    }
}
